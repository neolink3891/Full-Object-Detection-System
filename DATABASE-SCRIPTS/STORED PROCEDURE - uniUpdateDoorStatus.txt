CREATE DEFINER=`altadmin`@`%` PROCEDURE `altdb`.`uniUpdateDoorStatus`(
	IN v_did INT,
	IN v_isemp INT
)
BEGIN
	DECLARE v_exi INT;
	DECLARE v_res INT;

	DECLARE v_usu VARCHAR(50);
	DECLARE v_not VARCHAR(250);
	DECLARE v_rna VARCHAR(100);
	DECLARE v_tar VARCHAR(200);
	DECLARE v_fec VARCHAR(10);


	SELECT COUNT(1) INTO v_exi FROM UNI_LOCKDOORS WHERE DOORNUMBER = v_did;
	IF(v_exi = 1) THEN
		UPDATE UNI_LOCKDOORS SET ISEMPTY = v_isemp WHERE DOORNUMBER  = v_did;
		SELECT 1 AS 'SRES', 'OK' AS 'SMEN';
	ELSE
		SELECT -1 AS 'SRES', 'FAIL' AS 'SMEN';
	END IF;
END;
